def calc_total_percent(df):
  df['total'] = df['count'].sum()
  df['percent'] = df['count'] / df['total'] * 100
  return df
def calc_bienios(df, ano_bienio=2010):
  ano1 = ano_bienio if ano_bienio % 2 == 0 else ano_bienio - 1
  ano2 = ano_bienio + 1
  df_flt = df[(df['ano'] == ano1) | (df['ano'] == ano2)]
  cols_flt = df_flt.columns[1:-1]
  df_flt = df_flt[cols_flt].groupby(
    by=list(cols_flt[:-2]), as_index=False).sum()
  df_flt['percent'] = df_flt['count'] / df_flt['total'] * 100
  if len(df_flt) == 0:
    df_flt['count'] = [0]
    df_flt['percent'] = [0]
  return df_flt

def get_linreg_params(y):
  X = np.array([i + 1 for i, _ in enumerate(y)]).reshape(-1, 1)
  y = np.array(y)
  reg = LinearRegression().fit(X, y)
  angle = reg.coef_[0]
  intercept = reg.intercept_
  return angle, intercept
def dict_info(df, nome, proc, categ_name, categ_val):
  mean = df['percent'].mean()
  angle, intercept = get_linreg_params(df['percent'])
  df_bie_10 = calc_bienios(df)
  df_bie_18 = calc_bienios(df, 2018)
  return {
    'Nomeclatura': nome,
    'Tipo Parto': proc,
    'Categorização': categ_name,
    'Categoria': categ_val,
    'Procedimentos': df['count'].sum(),
    'Count Biênio 2010-11': df_bie_10['count'].values[0],
    'Count Biênio 2018-19': df_bie_18['count'].values[0],
    '% Biênio 2010-11': df_bie_10['percent'].values[0],
    '% Biênio 2018-19': df_bie_18['percent'].values[0],
    'Média Década 2010': mean,
    'Interceptação no Eixo Y': intercept,
    'Ângulo de Tendência': angle,
  }

Filtragem

def filter_by_nomeclatura(df, nome, ref='origem'):
  if (ref == 'origem') or (ref == 'destino'):
    df_filt = df[df[f'{ref}_nomeclatura'].str.contains(nome)]
  else:
    df_filt = df[(
      df['origem_nomeclatura'].str.contains(nome)
    ) | (
      df['destino_nomeclatura'].str.contains(nome))]
  return df_filt.copy()
def filter_by_procedimento(df, proc):
  return df[df['procedimento'] == proc].copy()
def filter_by_ano(df, ano):
  return df[(df['ano'] == ano)].copy()

def iter_anos(df):
  dfs_anos = list()
  for ano in df['ano'].unique():
    df_ = df[(df['ano'] == ano)].copy()
    df_['total'] = df_['count'].sum()
    df_['percent']  = df_['count'] / df_['total'] * 100
    dfs_anos.append(df_)
  return pd.concat(dfs_anos)
def iter_anos_deslc(df):
  dfs_anos = list()
  for ano in df['ano'].unique():
    df_ = df[(df['ano'] == ano)].copy()
    for cri in range(df_['criticidade'].max()):
      df_num = df_[df_['criticidade'] > cri]
      df_den = df_[df_['criticidade'] >= cri]
      d = {
        'ano': ano,
        'criticidade': f'(>{cri}) / (>={cri})',
        'count': df_num['count'].sum(),
        'total': df_den['count'].sum(),
        'percent': df_num['count'].sum() / df_den['count'].sum() * 100
      }
      dfs_anos.append(d)
  df_ano = pd.DataFrame(dfs_anos)
  return df_ano
def iter_procs(df, func_iter_anos):
  dfs_procs = list()
  for proc in df['procedimento'].unique():
    df_proc = filter_by_procedimento(df, proc)
    df_anos = func_iter_anos(df_proc)
    df_anos['procedimento'] = proc
    dfs_procs.append(df_anos)
  return pd.concat(dfs_procs)
def iter_infos(df, info_col, nome, categ_name):
  dfs_infos = list()
  for proc in df['procedimento'].unique():
    df_pr = filter_by_procedimento(df, proc)
    for categ in df_pr[info_col].unique():
      dfc = df_pr[df_pr[info_col] == categ]
      info = dict_info(dfc, nome, proc, categ_name, categ)
      dfs_infos.append(info)
  return pd.DataFrame(dfs_infos)

def trend_table(df, nome, info_col, categ_name, categ_func, func_iter_anos):
  cols_df = ['ano', 'procedimento', info_col, 'count']
  df_mdf = filter_by_nomeclatura(df, nome)
  if info_col == 'all': df_mdf[info_col] = 'procedimentos'
  df_mdf = df_mdf[cols_df].copy()
  df_mdf[info_col] = df_mdf[info_col].apply(categ_func)
  df_mdf = df_mdf.groupby(cols_df[:-1], as_index=False).sum()
  df_mdf = df_mdf.sort_values(by=cols_df[:-1])
  df_mdf = iter_procs(df_mdf, func_iter_anos)
  df_mdf = iter_infos(df_mdf, info_col, nome, categ_name)
  return df_mdf
def get_trend_table(df_proc, nome, categ):
  df = trend_table(df_proc, nome, *categorias[categ])
  return df
