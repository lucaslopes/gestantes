Venn

c = venn2(
  subsets = (
    n_sinasc - n_sinasc_sih,
    n_sih - n_sinasc_sih,
    n_sinasc_sih,
  ), set_labels = ('SINASC', 'SIH'),
  set_colors=('blue', 'yellow'))
c.get_patch_by_id('11').set_color('green')
c.get_patch_by_id('11').set_edgecolor('none')
c.get_patch_by_id('11').set_alpha(0.4)
plt.show()
