def add_info_cols(df, cols, ref):
  df_codes = df[:, cols]
  col_key = cols[0]
  if ref == 'destino':
    col_key = 'hosp_' + col_key
    df_codes.names = {cols[0]: col_key}
  df_codes.names = {c: f'{ref}_{c}' for c in cols[2:]}
  df_codes.names = {cols[1]: ref}
  df_codes.key = col_key
  return df_codes
