Gerando Tabela

def rank_locs(df_nom, path_locs, scale):
  cols_locs = {
    'nomeclatura': 'nomeclatura', f'nome_{scale}': 'nome', ...: None}
  df_locs = dt.fread(path_locs, columns=cols_locs)
  df_locs.key = 'nomeclatura'
  df_nmcl = df_nom[:, :, dt.join(df_locs)]
  escalas = {
    'pais': 1, 'regiao': 2, 'uf': 3,
    'regiao_saude': 4, 'municipio': 5, 'cnes': 6}
  pos_scale = escalas[scale]
  df_nmcl['nomeclatura'] = dt.Frame([
    '/'.join(c.split('/')[:pos_scale + 1]) + '/'
      for c in df_nmcl['nomeclatura'].to_list()[0]])
  df_nmcl['identificador'] = dt.Frame([
    c.split('/')[-2]
      for c in df_nmcl['nomeclatura'].to_list()[0]])
  df_nmcl['escala'] = scale
  cols = df_nmcl.names
  df_nmcl['count'] = 1
  df_nmcl = df_nmcl[:, dt.sum(f.count), by(df_nmcl.names[:-1])]
  df_nmcl = df_nmcl[:, f[:].remove(f.count)]
  return df_nmcl
